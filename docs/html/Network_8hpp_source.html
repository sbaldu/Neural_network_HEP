179Network<T, W, Activator, Loss>::Network(const std::vector<int>& nodes_per_layer)
180 : n_layers{static_cast<int>(nodes_per_layer.size())},
181 m_layers(n_layers + 1),
182 m_weights(n_layers + 1),
183 m_bias(n_layers - 1) {
184 for (int i{}; i < n_layers - 1; ++i) {
185 m_layers[i] = std::make_shared<Layer<T>>(nodes_per_layer[i]);
186 m_weights[i] = std::make_shared<Matrix<W>>(nodes_per_layer[i + 1], nodes_per_layer[i]);
187 m_bias[i] = std::make_shared<std::vector<W>>(nodes_per_layer[i + 1]);
188
189 // Generate random weight matrices
190 random_matrix(m_weights[i]);
191 }
192 m_layers[n_layers - 1] = std::make_shared<Layer<T>>(nodes_per_layer.back());
193 m_layers[n_layers] = nullptr;
194 m_weights[n_layers - 1] = nullptr;
195 m_weights[n_layers] = nullptr;
196}

319template <typename U>
320void Network<T, W, Activator, Loss>::back_propagation(const std::vector<U>& target,
321 int layer_id,
322 double eta) {
323 Loss<T, W, Activator> loss_function;
324 Matrix<W> loss_grad(loss_function.grad(target, layer_id + 1, m_layers, m_weights));
325 Matrix<T> activated_values_grad(m_layers[layer_id]->nodes());
326 *m_weights[layer_id] -= eta * (loss_grad * activated_values_grad.transpose());
327 *m_bias[layer_id] -= eta * loss_grad;
328}

374void Network<T, W, Activator, Loss>::import_network(const std::string& filepath) {
375 std::ifstream file_stream(filepath);
376
377 std::string file_row;
378 int i{};
379 while (getline(file_stream, file_row)) {
380 std::vector<W> weights;
381 std::vector<W> bias;
382
383 std::stringstream row_stream(file_row);
384 std::string value;
385 // First we fill the weight matrix
386 while (getline(row_stream, value, ',')) {
387 weights.push_back(std::stod(value));
388 }
389 // Then we get the next line and fill the bias vector
390 getline(file_stream, file_row);
391 row_stream = std::stringstream(file_row);
392 while (getline(row_stream, value, ',')) {
393 bias.push_back(std::stod(value));
394 }
395
396 m_weights[i] = std::make_shared<Matrix<W>>(m_weights[i]->nrows(), m_weights[i]->ncols(), weights);
397 m_bias[i] = std::make_shared<std::vector<W>>(bias);
398 }
399}
